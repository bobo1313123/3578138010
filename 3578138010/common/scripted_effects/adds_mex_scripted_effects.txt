MEX_junfang_jieru_increase = {
	custom_effect_tooltip = MEX_junfang_jieru_increase_tt
	custom_effect_tooltip = generic_skip_one_line_tt

	if = {
		limit = {
			has_idea = MEX_jundui_jieru_1
		}
		swap_ideas = {
			remove_idea = MEX_jundui_jieru_1
			add_idea = MEX_jundui_jieru_2
		}
	}
	else_if = {
		limit = {
			has_idea = MEX_jundui_jieru_2
		}
		swap_ideas = {
			remove_idea = MEX_jundui_jieru_2
			add_idea = MEX_jundui_jieru_3
		}
	}
	else_if = {
		limit = {
			has_idea = MEX_jundui_jieru_3
		}
		swap_ideas = {
			remove_idea = MEX_jundui_jieru_3
			add_idea = MEX_jundui_jieru_4
		}
	}
	else_if = {
		limit = {
			has_idea = MEX_jundui_jieru_4
		}
		swap_ideas = {
			remove_idea = MEX_jundui_jieru_4
			add_idea = MEX_jundui_jieru_5
		}
	}
	else_if = {
		limit = {
			has_idea = MEX_jundui_jieru_5
		}
		swap_ideas = {
			add_stability_factor = -0.05 
		}
	}
	else = {
		add_ideas = MEX_jundui_jieru_1
	}
}
